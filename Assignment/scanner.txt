%{
#include "arith.tab.h"
%}

%%
"var"                   { return VAR;      }
[a-zA-Z][a-zA-Z0-9_]*   { return IDENT;    }
":="                    { return ASSIGN;   }
"true"|"false"          { return BOOL;     }
[0-9]+	                { return NUM;      }
"["                     { return LSQBR;    }
"]"                     { return RSQBR;    }
","                     { return COMMA;    }
\".*\" | \'.*\'         { return STRING;   }
"*"	                    { return MUL;      }
"/"	                    { return DIV;      }
"%"	                    { return MOD;      }
"+"	                    { return ADD;      }
"-"	                    { return SUB;      }
"<"	                    { return LT;       }
"<="                    { return LTEQ;     }
">"	                    { return GT;       }
">="                    { return GTEQ;     }
"=="                    { return EQ;       }
"!="                    { return NEQ;      }
"~"                     { return BITNOT;   }
"<<"                    { return SAL;      }
">>"                    { return SAR;      }
"&"                     { return BITAND;   }
"^"                     { return BITXOR;   }
"|"                     { return BITOR;    }
"not"                   { return NOT;      }
"and"                   { return AND;      }
"or"                    { return OR;       }
"function"              { return FUNCTION; }
"("                     { return LPAREN;   }
")"                     { return RPAREN;   }
"begin"                 { return BEGIN;    }
"end"                   { return END;      }
"if"                    { return IF;       }
"then"                  { return THEN;     }
"else"                  { return ELSE;     }
"while"                 { return WHILE;    }
"do"                    { return DO;       }
"for"                   { return FOR;      }
"to"                    { return TO;       }
\n                      { return EOL;      }

"//".*                  { /* ignore comments */ }
[\s]                    { /* ignore white space */ }
.	                    { yyerror("Mystery character %c\n", *yytext); }
%%



// same as above but syntax is changed to work with https://bottlecaps.de/rr/ui


VAR                 ::=                "v" "a" "r"
IDENT               ::=                [a-zA-Z][a-zA-Z0-9_]*
ASSIGN              ::=                ":" "="
BOOL                ::=                "t" "r" "u" "e"|"f" "a" "l" "s" "e"
NUM                 ::=                [0-9]+
LSQBR               ::=                "["
RSQBR               ::=                "]"
COMMA               ::=                ","
STRING              ::=                '"' [^"]* '"' | "'" [^']* "'"
MUL                 ::=                "*"
DIV                 ::=                "/"
MOD                 ::=                "%"
ADD                 ::=                "+"
SUB                 ::=                "-"
LT                  ::=                "<"
LTEQ                ::=                "<" "="
GT                  ::=                ">" 
GTEQ                ::=                ">" "="
EQ                  ::=                "=" "="
NEQ                 ::=                "!" "="
BITNOT              ::=                "~"
SAL                 ::=                "<" "<"
SAR                 ::=                ">" ">"
BITAND              ::=                "&"
BITXOR              ::=                "^"
BITOR               ::=                "|"
NOT                 ::=                "n" "o" "t"
AND                 ::=                "a" "n" "d"
OR                  ::=                "o" "r"
FUNCTION            ::=                "f" "u" "n" "c" "t" "i" "o" "n"
LPAREN              ::=                "("
RPAREN              ::=                ")"
BEGIN               ::=                "b" "e" "g" "i" "n"
END                 ::=                "e" "n" "d"
IF                  ::=                "i" "f"
THEN                ::=                "t" "h" "e" "n"
ELSE                ::=                "e" "l" "s" "e"
WHILE               ::=                "w" "h" "i" "l" "e"
DO                  ::=                "d" "o"
FOR                 ::=                "f" "o" "r"
TO                  ::=                "t" "o"
EOL                 ::=                ";"
